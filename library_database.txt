#create database

create database library;
use library;

#create tables

create table USERS (
    userName varchar(25) NOT NULL,
    password varchar(100) NOT NULL
)
ENGINE InnoDB;

create table BOOKS (
	bookId int(11) NOT NULL AUTO_INCREMENT,
	bookName varchar(100) NOT NULL,
	author varchar(100) NOT NULL,
	genre varchar(100) NOT NULL,
	publishYear varchar(100) NOT NULL,
	imageLink varchar(100) NOT NULL,
	PRIMARY KEY (bookId)
)
ENGINE InnoDB;

create table MAGAZINES (
	magazineId int(11) NOT NULL AUTO_INCREMENT,
	magazineName varchar(100) NOT NULL,
	topic varchar(100) NOT NULL,
	publisher varchar(100) NOT NULL,
	publishYear varchar(100) NOT NULL,
	issueNumber int(50) NOT NULL,
	imageLink varchar(100) NOT NULL,
	PRIMARY KEY (magazineId)
)
ENGINE InnoDB;

create table MOVIES (
	movieId int(11) NOT NULL AUTO_INCREMENT,
	movieName varchar(100) NOT NULL,
	genre varchar(100) NOT NULL,
	director varchar(100) NOT NULL,
	releaseYear varchar(100) NOT NULL,
	imageLink varchar(100) NOT NULL,
	PRIMARY KEY (movieId)
)
ENGINE InnoDB;

create table MUSIC (
	musicId int(11) NOT NULL AUTO_INCREMENT,
	artistName varchar(100) NOT NULL,
	albumName varchar(100) NOT NULL,
	genre varchar(100) NOT NULL,
	releaseYear varchar(100) NOT NULL,
	imageLink varchar(100) NOT NULL,
	PRIMARY KEY (musicId)
)
ENGINE InnoDB;
